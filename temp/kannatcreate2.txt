);

CREATE TABLE tuote
(
tuote_id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
tuote_nimi VARCHAR(30) NOT NULL,
tuote_hinta DECIMAL(4,2) NOT NULL,
FOREING KEY (tryhma_id) REFERENCES tuoteryhma(tryhma_id)
);

CREATE TABLE tuotteentaytteet
(
tuottay_id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
FOREING KEY (tayte_id) REFERENCES tayte(tayte_id),
FOREING KEY (tuote_id) REFERENCES tuote(tuote_id)
);

CREATE TABLE tilaus
(
tilaus_id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
tilausosoite VARCHAR(255) NOT NULL,
tilauspvm TIMESTAMP,
hintasumma DECIMAL(6,2),
etun_asiakas VARCHAR(18) NOT NULL,
sukun_asiakas VARCHAR(30) NOT NULL,
puh_asiakas
);

CREATE TABLE tilausrivi
(
tilriv_id int NOT NULL PRIMARYKEY AUTO_INCREMENT,
maara INTEGER NOT NULL,
rivihinta DECIMAL(6,2),
oregano BOOLEAN,
PRIMARY KEY (tilriv_id),
FOREING KEY (tilaus_id) REFERENCES tilaus(tilaus_id)
);
