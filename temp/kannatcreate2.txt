CREATE TABLE tayte
(
tayte_id int NOT NULL AUTO_INCREMENT,
tayte_nimi VARCHAR(30) NOT NULL,
PRIMARY KEY (tayte_id)
)

CREATE TABLE tuoteryhma
(
tryhma_id int NOT NULL AUTO_INCREMENT,
tryhma_nimi VARCHAR(8) NOT NULL,
PRIMARY KEY (tryhma_id)
)

CREATE TABLE tuote
(
tuote_id int NOT NULL AUTO_INCREMENT,
tuote_nimi VARCHAR(30) NOT NULL,
tuote_hinta 
tryhma_id
PRIMARY KEY (tuote_id),
FOREING KEY (tryhma_id) REFERENCES tuoteryhma(tryhma_id)
)

CREATE TABLE tuotteentaytteet
(
tuottay_id int NOT NULL AUTO_INCREMENT,
PRIMARY KEY (tuottay_id),
FOREING KEY (tayte_id) REFERENCES tayte(tayte_id),
FOREING KEY (tuote_id) REFERENCES tuote(tuote_id)
)

(
tilaus_id INT NOT NULL AUTO_INCREMENT,

tilauspvm
hintasumma
etun_asiakas VARCHAR(18) NOT NULL,

puh_asiakas

)

(
tilriv_id int NOT NULL AUTO_INCREMENT,
